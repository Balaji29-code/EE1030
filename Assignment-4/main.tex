    a &= BC = 7cm \\
    b &= AC \\
    c &= AB \\
    \therefore b + c = 13cm \\
    \since b^2 = a^2 +c^2 - 2ac\cosB \\
    \implies (13 - c)^2 = a^2 - 2ac\cosB \\
